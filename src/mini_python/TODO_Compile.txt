#################################################################"
TYPES
#################################################################"

== INT
%rdi = address to value in memory
TDA     | integer
8 bytes | 8 bytes

== BOOL
%rdi = address to value in memory
TDA     | padding | value
8 bytes | 7 bits  | 1 bit

== STRING
%rdi = address to value in memory
TDA     | padding | value
8 bytes | 7 bits  | 1 bit

== LIST
%rdi = address to value in memory
TDA     | length  | addr1   | addr2   | ... |
8 bytes | 8 bytes | 8 bytes | 8 bytes | ... |


LEAQ D(%x, %y, k), %r
%r = %r + %y * k
k=1,2,4,8
